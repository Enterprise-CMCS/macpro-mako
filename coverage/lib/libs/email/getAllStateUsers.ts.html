export const getAllStateUsers = async ({
  userPoolId,
  state,
}: {
  userPoolId: string;
  state: string;
}): Promise<StateUser[]> => {
  try {

    const command = new ListUsersCommand(params);

      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
